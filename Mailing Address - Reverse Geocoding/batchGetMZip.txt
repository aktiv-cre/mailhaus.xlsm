Sub vbaMacroBatchGetMailingZip()
    Dim lastRow As Long
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Master Sheet") ' change this according to the name of sheet

    lastRow = ws.Cells(ws.Rows.Count, "AG").End(xlUp).Row ' Find the last row based on Street column (AG)
    
    Dim i As Long
    For i = 2 To lastRow ' Starting from row 2 (assuming row 1 is headers)
        ' Call GetPostalCodeFromParsedAddress with Street, City, and State
        ws.Cells(i, "AQ").Value = GetPostalCodeFromParsedAddress(ws.Cells(i, "AG").Value, ws.Cells(i, "AM").Value, ws.Cells(i, "AO").Value) ' these are the reference cells (mailing street 1, state, zip)
    Next i

    MsgBox "Zip codes applied to all rows in column AQ."
End Sub
